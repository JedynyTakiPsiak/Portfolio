let
    Source = Pdf.Tables(File.Contents("TUTAJ\WKLEIC\NASZA\SCIEZKA\PLIKU.pdf"), [Implementation="1.3"]),
    TablesOnly = Table.SelectRows(Source, each ([Kind] = "Table")),
    Expanded = Table.Combine(TablesOnly[Data]),

    AddColList = Table.AddColumn(Expanded, "ColList", each Record.FieldNames(_)),
    OnlyWithData = Table.SelectRows(AddColList, each 
        List.Contains([ColList], "Column3") and List.Contains([ColList], "Column4") and [Column3] <> null and [Column4] <> null
    ),
    RemoveColList = Table.RemoveColumns(OnlyWithData, {"ColList"}),

    // Konwersja wartości tekstowych w Column4 do liczby
    kWhText = Table.TransformColumns(
        RemoveColList, 
        {
            {"Column4", each Text.Trim(Text.Clean(Text.Replace(_, ",", "."))), type text}
        }
    ),
    kWhToNumber = Table.TransformColumnTypes(
        kWhText,
        {{"Column4", type number}},
        "en-US"
    ),

    // Naprawianie Column3 — konwersja różnych typów na datetime
    ConvertedDates = Table.TransformColumns(
        kWhToNumber,
        {
            {"Column3", each 
                if Value.Is(_, type number) then 
                    #datetime(1899, 12, 30, 0, 0, 0) + Duration.From(_) 
                else if Value.Is(_, type text) then 
                    try DateTime.From(_) otherwise null 
                else _, 
            type datetime}
        }
    ),

    // Filtrowanie niepustych
    Cleaned = Table.SelectRows(ConvertedDates, each [Column3] <> null and [Column4] <> null),

    // Zaokrąglenie w dół do pełnej godziny z korektą 15 min
    Hour = Table.AddColumn(
        Cleaned, 
        "Godzina", 
        each Time.StartOfHour([Column3] - #duration(0, 0, 15, 0)), 
        type datetime
    ),

    // Kolumna tekstowa do eksportu do Excela
    Formatted = Table.AddColumn(
        Hour, 
        "Data i czas (tekst)", 
        each DateTime.ToText([Column3], "dd.MM.yyyy HH:mm"), 
        type text
    ),

    // Grupowanie po godzinie
    Grouped = Table.Group(
        Formatted, 
        {"Godzina"}, 
        {{"Zużycie [kWh]", each List.Sum([Column4]), type number}}
    ),

    // Sortowanie po czasie
    Sorted = Table.Sort(Grouped,{{"Godzina", Order.Ascending}})
in
    Sorted
